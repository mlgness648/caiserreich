	id = tram.1

	title = tram.1.t
	desc = tram.1.d
	is_triggered_only = yes

	option = { #war
		name = tram.1.a
		ai_chance = {
			factor = 70
		}
		set_global_flag = PRI_breaksaway
		hidden_effect = {
			TRM = {
				remove_ideas = TRM_JAP_INF_START
				remove_ideas = TRM_JAP_INF_TRADE
				remove_ideas = TRM_JAP_DEP_1
				remove_ideas = TRM_JAP_DEP_2
				remove_ideas = TRM_JAP_DEP_3
				remove_ideas = TRM_JAP_NAV_MIS
				remove_ideas = TRM_JAP_MIL_MIS
				remove_ideas = TRM_JAP_CONS
				remove_ideas = hino_automobile
				remove_ideas = nakajima_factory_vladivostok
				separate_peace_all_wars = yes
			}
		}
		JAP = {
			remove_from_faction = TRM

		}
	}

	option = { #leave them
		name = tram.1.b
		ai_chance = {
			factor = 30
		}
		set_global_flag = PRI_breaksaway
		add_stability = -0.15
		hidden_effect = {
			TRM = {
				remove_ideas = TRM_JAP_INF_START
				remove_ideas = TRM_JAP_INF_TRADE
				remove_ideas = TRM_JAP_DEP_1
