# Cosmetic Change for GEX
country_event = {
	id = gerfla.11
	immediate = { log = "[GetDateText]: [Root.GetName]: event gerfla.11" }
	title = gerfla.11.t
	desc = gerfla.11.d
	picture = GFX_report_event_german_reichstag_gathering

	fire_only_once = yes

	trigger = {
		tag = GEX
		NOT = {
			has_cosmetic_tag = GER
		}
		OR = {
			64 = {
				is_owned_and_controlled_by = ROOT
				} #Brandenburg - Berlin
			55 = {
				is_owned_and_controlled_by = ROOT
				} #Hessen - Frankfurt
			59 = {
				is_owned_and_controlled_by = ROOT
				} #Hannover - Hamburg
			51 = {
				is_owned_and_controlled_by = ROOT
				} #Ostrheinland - Essen
			52 = {
				is_owned_and_controlled_by = ROOT
				} #Oberbayern - Munich
		}
	}

	mean_time_to_happen = {
		days = 3
	}

	option = {
		name = gerfla.11.a
		add_political_power = 50
		add_stability = 0.05
		hidden_effect = {
			set_cosmetic_tag = GER
		}
		custom_effect_tooltip = GEX_becomes_GER

		if = {
			limit = {
				64 = {
					is_owned_and_controlled_by = ROOT
				}
			}
			set_capital = 64
		}
		else_if = {
			limit = {
				55 = {
					is_owned_and_controlled_by = ROOT
				}
			}
			set_capital = 55
		}
		else_if = {
			limit = {
				59 = {
					is_owned_and_controlled_by = ROOT
				}
			}
			set_capital = 59
		}
		else_if = {
			limit = {
				51 = {
					is_owned_and_controlled_by = ROOT
				}
			}
			set_capital = 51
		}
		else = {
			set_capital = 52
		}
		if = {
			limit = { is_in_array = { global.african_tags_array = GEX } }
			remove_from_array = { global.african_tags_array = GEX }
			add_to_array = { global.european_tags_array = GEX }
		}
	}
}
