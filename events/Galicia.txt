add_namespace = gal

country_event = { #Galicia becomes independent, after war
	id = gal.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event gal.1" }
	title = gal.1.t
	desc = gal.1.d

	trigger = {
		tag = GAL
		has_war = no
		is_subject = no
		10 = {
			owner = { is_polish_tag = yes }
		}
	}
	mean_time_to_happen = {
		days = 14
	}

	fire_only_once = yes

	option = { #join Poland
		name = gal.1.a
		ai_chance = {
			factor = 80
		}
		add_political_power = -100
		10 = {
			owner = {
				country_event = {
					id = gal.2
					days = 2
				}
			}
		}
	}
	option = { #stay independent
		name = gal.1.b
		ai_chance = {
			factor = 20
		}
		add_political_power = -100
	}
}

country_event = { #Galicia wants to join Poland
	id = gal.2
	immediate = { log = "[GetDateText]: [Root.GetName]: event gal.2" }
	title = gal.2.t
	desc = gal.2.d

	is_triggered_only = yes
	fire_only_once = yes

	option = { #excellent
		name = gal.2.a
		add_political_power = -30

		annex_country = {
			target = GAL
			transfer_troops = yes
		}

		if = {
			limit = {
				owns_state = 713
				ROM = {
					exists = yes
					has_war = no
				}
			}
			ROM = {
				country_event = {
					id = gal.3
					days = 1
				}
			}
		}

		if = {
			limit = {
				owns_state = 713
				RRM = {
					exists = yes
					has_war = no
				}
			}
			RRM = {
				country_event = {
					id = gal.3
					days = 1
				}
			}
		}

		if = {
			limit = {
				OR = {
					owns_state = 73
					owns_state = 80
					owns_state = 89
					owns_state = 91
				}
				UKR = {
					exists = yes
					has_war = no
				}
			}
			UKR = {
				country_event = {
					id = gal.4
					days = 3
				}
			}
		}
	}
}
