# Hidden event to trigger The White Terror comes to <state>
country_event = {
	id = royalist.16

	trigger = {
		tag = RRM
		NOT = {
			country_exists = ROM
		}

		OR = {
			has_idea = RRM_chaos_low
			has_idea = RRM_chaos_medium
			has_idea = RRM_chaos_high
			has_idea = RRM_anarchy
		}

		any_owned_state = {
			is_controlled_by = RRM
			NOT = {
				has_state_flag = white_terror
			}
		}
	}

	mean_time_to_happen = {
		days = 14
	}

	hidden = yes

	immediate = {
		log = "[GetDateText]: [Root.GetName]: event royalist.16"
		random_owned_controlled_state = {
			limit = {
				NOT = {
					has_state_flag = white_terror
				}
			}

			set_state_flag = white_terror

			state_event = {
				id = royalist.17
				trigger_for = controller
			}
		}
	}
}

# First Elections
country_event = {
	id = royalist.19
	title = royalist.19.t
	desc = {
		trigger = {
			OR = {
				has_idea = RRM_chaos_high
				has_idea = RRM_anarchy
			}
		}
		text = royalist.19.desc_chaos
	}
	desc = {
		trigger = {
			NOT = {
				has_idea = RRM_chaos_high
				has_idea = RRM_anarchy
			}
		}
		text = royalist.19.desc_no_chaos
	}
	picture = GFX_report_event_election_vote

	trigger = {
		tag = RRM

		has_country_flag = RRM_elections

		NOT = {
			has_country_flag = RRM_had_first_elections
		}
	}

	immediate = {
		log = "[GetDateText]: [Root.GetName]: event royalist.19"
		set_country_flag = RRM_had_first_elections
	}

	is_triggered_only = yes

	option = {
		name = royalist.19.a # PNT wins majority

		trigger = {
			NOT = {
				OR = {
					has_idea = RRM_chaos_high
					has_idea = RRM_anarchy
				}
			}
			OR = {
				has_government = social_conservative
				check_variable = {
					ROOT.party_popularity@social_conservative > ROOT.party_popularity@market_liberal
				}
			}
		}

		set_politics = {
			ruling_party = social_conservative
			election_frequency = 48
			elections_allowed = yes
		}

		# TODO set ministers

		add_popularity = {
			ideology = social_conservative
			popularity = 0.2
		}

		ai_chance = {
			factor = 40
		}
	}

	option = {
		name = royalist.19.b # PNL wins majority

		trigger = {
			OR = {
				has_government = market_liberal
				check_variable = {

			ideology = market_liberal

			factor = 40
		}
	}

			NOT = {
				has_government = social_conservative
				check_variable = {
					ROOT.party_popularity@social_conservative > ROOT.party_popularity@market_liberal
				}
			}
		}

			ruling_party = social_conservative
