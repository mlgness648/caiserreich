# Recurring Black monday instability
country_event = {
	id = libya.11
	immediate = { log = "[GetDateText]: [Root.GetName]: event libya.11" }
	title = libya.11.t
	desc = libya.11.d

	is_triggered_only = yes

	trigger = {
		tag = LBA
		has_country_flag = LBA_black_monday
		NOT = {
			has_completed_focus = LBA_black_monday_2
		}
	}

	option = {
		name = libya.11.a
		add_political_power = -20
		if = {
			limit = {
				NOT = {
					has_completed_focus = LBA_black_monday_2
				}
			}
			hidden_effect = {
				set_temp_variable = { test = random }
				multiply_temp_variable = { test = 100 }
				if = {
					limit = {
						check_variable = {
							test < 33
						}
					}
					country_event = {
						id = libya.11
						days = 60
					}
				}
				else_if = {
					limit = {
						check_variable = {
							test < 66
						}
					}
					country_event = {
						id = libya.11
						days = 100
					}
				}
				else = {
					country_event = {
						id = libya.11
						days = 140
					}
				}
			}
		}
	}
}
